```typescript
  isCompact?: boolean;
```

## 5. Zustand Store Standards

### 5.1 Store File Structure

```typescript
// /stores/usePostStore.ts

import { create } from 'zustand';
import { persist, createJSONStorage } from 'zustand/middleware';
import { MMKVStorage } from '@/config/storage';  // Unified MMKV adapter
import type { Post } from '@/types/post.types';
import type { ServiceResult } from '@/types/common.types';
import { PostService } from '@/services/post.service';

// 1. State type (data only, no actions)
interface PostState {
  posts: Post[];
  isLoading: boolean;
  error: string | null;
  currentPage: number;
}

// 2. Action type (methods only)
interface PostActions {
  fetchPosts: (userId: string) => Promise<void>;
  likePost: (postId: string) => Promise<void>;
  reset: () => void;
}

// 3. Initial state (standalone constant for easy reset)
const initialState: PostState = {
  posts: [],
  isLoading: false,
  error: null,
  currentPage: 1,
};

// 4. Store definition (named export)
export const usePostStore = create<PostState & PostActions>()(
  persist(
    (set, get) => ({
      ...initialState,

      fetchPosts: async (userId) => {
        set({ isLoading: true, error: null });
        const result = await PostService.getByUser(userId);
        if (result.error) {
          set({ error: result.error.message, isLoading: false });
          return;
        }
        set({ posts: result.data ?? [], isLoading: false });
      },

      likePost: async (postId) => {
        // Optimistic update
        const prevPosts = get().posts;
        set({ posts: prevPosts.map(p =>
          p.id === postId ? { ...p, likesCount: p.likesCount + 1 } : p
        )});
        const result = await PostService.like(postId);
        if (result.error) {
          set({ posts: prevPosts }); // Rollback
        }
      },

      reset: () => set(initialState),
    }),
    {
      name: 'post-store',
      storage: createJSONStorage(() => MMKVStorage),
      partialize: (state) => ({ posts: state.posts }), // Only persist posts, not loading/error
    }
  )
);
```

### 5.2 Store Naming Rules

| Rule | Example |
|------|---------|
| File name: use + EntityName + Store | `useUserStore.ts` |
| Loading state: isLoading (boolean) | `isLoading: false` |
| Error state: error (string \| null) | `error: null` |
| Every Store MUST include reset() | For clearing all data on sign-out |
| Action naming: verb + noun (camelCase) | `fetchPosts` `likePost` `deleteComment` |

## 6. Service Layer Standards

### 6.1 Service File Structure

```typescript
// /services/post.service.ts

import { supabase } from '@/config/supabase';
import type { Post } from '@/types/post.types';
import type { ServiceResult } from '@/types/common.types';

// All Service functions:
// - Return ServiceResult<T>; throw is FORBIDDEN
// - Parameters use explicit types; any is FORBIDDEN
// - Function name: verb + noun

export const PostService = {
  async getByUser(userId: string): Promise<ServiceResult<Post[]>> {
    const { data, error } = await supabase
      .from('posts')
      .select('id, content, likes_count, created_at, user:profiles(username, avatar_url)')
      .eq('user_id', userId)
      .order('created_at', { ascending: false })
      .limit(20);

    if (error) {
      return { data: null, error: { code: 'BIZ_POST_001', message: error.message } };
    }
    return { data: data as Post[], error: null };
  },

  async like(postId: string): Promise<ServiceResult<void>> {
    const { error } = await supabase.rpc('increment_likes', { post_id: postId });
    if (error) {
      return { data: null, error: { code: 'BIZ_POST_002', message: 'Like failed. Please try again.' } };
    }
    return { data: null, error: null };
  },
};
```

### 6.2 Service Layer Rules

```typescript
// ❌ DON'T — throw Supabase errors directly to upper layers
const { data, error } = await supabase.from('posts').select('*');
if (error) throw error; // FORBIDDEN

// ❌ DON'T — access supabase directly in components or Stores
const { data } = await supabase.from('posts').select(); // FORBIDDEN

// ✅ DO — all Supabase access goes through the Service layer, with a unified error format
const result = await PostService.getByUser(userId);
if (result.error) { /* handle standardized error */ }
```

## 9. Error Handling Standards

### 9.1 Error Handling Layers

```
Service Layer: Catch Supabase / network errors → Convert to ServiceResult.error (unified format)
Store Layer: Receive ServiceResult.error → Write to store.error (string) → Trigger UI response
Component Layer: Read store.error → Display user-friendly message (Snackbar / Alert)
Global Layer: ErrorBoundary catches render crashes → Display degraded UI
```

### 9.2 User Message Standards

